function nlml(θ)
    A, l, P, μ, σ² = unpack(θ)
    k = A * stretch(EQ(), 1 / l^2) * Stheno.Cosine(P / 2)
    g = GP(μ, k, GPC())
    return -logpdf(g(t, σ²), f)
end
